### 접근법

DFS에서 탈출조건을 설정하는데에 착오가 있어, 겁나 애먹었다.

### 얻어갈 점

1. ```to_string()```

int -> string 간의 변환은 to_string()을 통해 바로 가능하다. String은 char list와 접근 방식이 동일하기 때문에 char list에서 string으로 변환하였다.

반대의 변환은 ```stoi(n)```을 통해 가능하다.

2. Memoization

위 문제의 경우, 캐싱할 때 cnt 값까지 같이 저장해야 한다. (cnt 가 다른 경우 숫자가 달라도 경우가 달라짐.)

## Solution Code

```cpp
#include <iostream>
#include <string>
#include <vector>
#define MAX (8)

using namespace std;

void dfs(int idx, int cnt);
void solve();
void exchange(int a, int b);
bool isCached(int cnt);

int N, K;
string numstr;
vector<pair<string, int> > v; // cache
int ans = -1;

int main() {
    freopen("input.txt", "r", stdin);

    cin >> N >> K;
    numstr = to_string(N);

    solve();

    cout << ans << "\n";
}

void solve() {
    dfs(0, K);
}

void dfs(int idx, int cnt) {
    if (cnt == 0) {
        if (stoi(numstr) > ans) {
            ans = stoi(numstr);
        }
        return;
    } else if (idx >= numstr.size()) {
        return;
    }
    
    for (int i = 0; i < numstr.size(); i++) {
        if (idx == i) {
            continue;
        } else if ((idx == 0 && numstr[i] == '0') || i == 0 && numstr[idx] == '0') {
            continue;
        }
        exchange(idx, i);

        // Cache에 있는지 여부 확인
        if (isCached(cnt)) {
            exchange(idx, i);
            continue;
        }
        v.push_back(make_pair(numstr, cnt));
        dfs(idx, cnt - 1);
        exchange(idx, i);
    }
    dfs(idx + 1, cnt);
}

void exchange(int a, int b) {
    char temp = numstr[a];
    numstr[a] = numstr[b];
    numstr[b] = temp;
}

bool isCached(int cnt) {
    for (auto i : v) {
        if (i.first == numstr && i.second == cnt) {
            return true;
        }
    }
    return false;
}
```
