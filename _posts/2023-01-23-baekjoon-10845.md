기본적으로 큐은 deque primary STL을 이용해 구현되어 있으므로, 마찬가지로 deque을 이용함.

## Solution Code

```cpp
#include <iostream>
#include <string>
#include <deque>

using namespace std;
void push(int x);
void pop();
void size();
void empty();
void front();
void back();

int N;
deque<int> queue;

int main() {
    freopen("input.txt", "r", stdin);

    cin >> N;
    for (int i = 0; i < N; i++) {
        string c;
        cin >> c;
        if (c == "push") {
            int x;
            cin >> x;
            push(x);
        } else if (c == "pop") {
            pop();
        } else if (c == "size") {
            size();
        } else if (c == "empty") {
            empty();
        } else if (c == "front") {
            front();
        } else if (c == "back") {
            back();
        }
    }
}

void push(int x) {
    queue.push_back(x);
}

void pop() {
    if (queue.size() == 0) {
        cout << -1 << "\n";
        return;
    }

    int n = queue.front();
    queue.pop_front();
    cout << n << "\n";
}

void size() {
    cout << queue.size() << "\n";
}

void empty() {
    if (queue.size() == 0) {
        cout << 1 << "\n";
    } else {
        cout << 0 << "\n";
    }
}

void front() {
    if (queue.size() == 0) {
        cout << -1 << "\n";
        return;
    }

    int n = queue.front();
    cout << n << "\n";
}

void back() {
    if (queue.size() == 0) {
        cout << -1 << "\n";
        return;
    }

    int n = queue.back();
    cout << n << "\n";
}
```
