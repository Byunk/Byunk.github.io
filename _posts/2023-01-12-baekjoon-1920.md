### Missed Points

1. Binary Search 탈출조건에서 segmentation fault 지속적으로 발생함.

원소의 개수가 2개일 때, `floor((end - start) / 2)`로 조건을 탈출할 수 없음을 고려하지 않음.

### 얻어갈 점

## Solution Code

```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>

using namespace std;

void solve();
void find(int x, int start, int end);

int N, M;
vector<int> num;
vector<int> test;
vector<int> ans;

int main() {
    freopen("input.txt", "r", stdin);

    cin >> N;
    for (int i = 0; i < N; i++) {
        int n;
        cin >> n;
        num.push_back(n);
    }
    cin >> M;
    for (int i = 0; i < M; i++) {
        int n;
        cin >> n;
        test.push_back(n);
    }

    solve();

    for (int i = 0; i < M; i++) {
        cout << ans[i] << "\n";
    }
}

void solve() {
    // nlogn sort
    sort(num.begin(), num.end());

    for (int i = 0; i < M; i++) {
        find(test[i], 0, N-1);
    }
}

void find(int x, int start, int end) {
    int idx = start + floor((end - start) / 2);
    if (abs(start - end) <= 1) {
        if (num[start] == x || num[end] == x) {
            ans.push_back(1);
            return;
        } else {
            ans.push_back(0);
            return;
        }
    }

    if (num[idx] > x) {
        find(x, start, idx);
    } else if (num[idx] < x) {
        find(x, idx, end);
    } else {
        ans.push_back(1);
        return;
    }
}
```
