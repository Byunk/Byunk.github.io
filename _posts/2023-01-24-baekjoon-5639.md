```cpp
#include <iostream>
#include <vector>

using namespace std;

void solve();

vector<int> v;
vector<int> out;

struct node {
    int key;
    node* left;
    node* right;
};

struct BST {
    node* root;

    static BST create(int key) {
        BST bst;
        bst.root = new node {key, NULL, NULL};
        return bst;
    }

    static node* search(node* node, int key) {
        if (node == NULL) {
            return NULL;
        }

        if (node->key == key) {
            return node;
        }

        auto leftSearch = search(node->left, key);

        if (leftSearch != NULL) {
            return leftSearch;
        }

        return search(node->right, key);
    }

    void add(node* n, int key) {
        if (n == NULL) {
            return;
        }

        if (key < n->key) {
            if (!n->left) {
                n->left = new node{key, NULL, NULL};
                return;
            }
            add(n->left, key);
        } else {
            if (!n->right) {
                n->right = new node{key, NULL, NULL};
                return;
            }
            add(n->right, key);
        }
    }

    void postSearch(node* node) {
        if (node == NULL) {
            return;
        }

        postSearch(node->left);
        postSearch(node->right);
        out.push_back(node->key);
    }
};

int main() {
    freopen("input.txt", "r", stdin);

    int n;
    while(cin >> n) {
        v.push_back(n);
    }

    solve();

    for (int i = 0; i < out.size(); i++) {
        cout << out[i] << "\n";
    }
}

void solve() {
    BST bst;
    for (int i = 0; i < v.size(); i++) {
        if (i == 0) {
            bst = BST::create(v[i]);
            continue;
        }
        bst.add(bst.root, v[i]);
    }
    bst.postSearch(bst.root);
}
```
