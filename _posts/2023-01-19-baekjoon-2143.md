## Recap

### 접근법

시간복잡도 계산이 중요한 문제. 모든 부분합의 경우의 수를 계산한는 과정이 n!인데, n이 1000이하 이므로, 대략 1000000회가 걸린다. 이를 이용해, 모든 부분집합의 경우의 수를 구하고, 정렬하여 (nlogn) two pointer로 비교한다. 마지막에 binary search를 이용해도 될듯?

### 시간복잡도

O(n^2) + O(nlogn) + O(n)

### 공간복잡도

### Missed Points

1. 시간복잡도 추산에 오류가 있었음.
2. 문제 조건에 의해 ans의 type이 long long이어야 하는데, 이를 놓쳤음.
3. Two pointer algorithm에서 subA의 원소와 subB의 원소의 합이 T와 같을 때 경우의 수를 구하는 로직에서 큰 오류가 있어, 계속 실패했다.

아래와 같이 같은 경우, 반복문을 돌면서 모든 같은 경우를 구하고 두 경우의 수를 곱해줘야 했는데, 둘 중 하나를 단순히 increment하는 바람에 오류가 발생했다.

```cpp
long long acnt, bcnt;
acnt = bcnt = 1;
while (a < subA.size() - 1 && subA[a+1] == subA[a]) {
    acnt++;
    a++;
}
while (b < subB.size() - 1 && subB[b+1] == subB[b]) {
    bcnt++;
    b++;
}
ans += acnt * bcnt;
```

### 얻어갈 점

## Solution Code

```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#define MAX 1001

using namespace std;

void solve();

int T, N, M;
int A[MAX];
int B[MAX];
vector<long long> subA;
vector<long long> subB;

long long ans;

int main() {
    freopen("input.txt", "r", stdin);

    cin >> T;
    cin >> N;
    for (int i = 0; i < N; i++) {
        cin >> A[i];
    }
    cin >> M;
    for (int i = 0; i < M; i++) {
        cin >> B[i];
    }

    solve();

    cout << ans << "\n";
}

void solve() {
    // 모든 부분합 구하기
    for (int i = 0; i < N; i++) {
        long long sum = 0;
        for (int j = i; j < N; j++) {
            sum += A[j];
            subA.push_back(sum);
        }
    }

    for (int i = 0; i < M; i++) {
        long long sum = 0;
        for (int j = i; j < M; j++) {
            sum += B[j];
            subB.push_back(sum);
        }
    }

    // 정렬
    sort(subA.begin(), subA.end(), greater<long long>());
    sort(subB.begin(), subB.end());

    // 경우의 수 구하기
    int a, b;
    a = b = 0;
    long long sum;

    while (true) {
        if (a == subA.size() || b == subB.size()) {
            break;
        }

        sum = subA[a] + subB[b];

        if (sum > T) {
            a++;
        } else if (sum < T) {
            b++;            
        } else {
            //*
            long long acnt, bcnt;
            acnt = bcnt = 1;
            while (a < subA.size() - 1 && subA[a+1] == subA[a]) {
                acnt++;
                a++;
            }
            while (b < subB.size() - 1 && subB[b+1] == subB[b]) {
                bcnt++;
                b++;
            }
            ans += acnt * bcnt;
            if (a != subA.size()-1) {
                a++;
            } else {
                b++;
            }
        }
    }
}

```
