## Solution Code

```cpp
#include <iostream>
#include <queue>

#define MAX 51

using namespace std;

void solve();

typedef pair<int, int> pii;

int R, C;
char map[MAX][MAX];
bool visited[MAX][MAX];
queue<pii> waterQ, q;

int dr[4] = {-1, 1, 0, 0};
int dc[4] = {0, 0, -1, 1};

int ans = -1;

int main() {
    // freopen("input.txt", "r", stdin);

    cin >> R >> C;
    for (int i = 0; i < R; i++) {
        for (int j = 0; j < C; j++) {
            cin >> map[i][j];
        }
    }
    solve();
    if (ans == -1) {
        cout << "KAKTUS\n";
    } else {
        cout << ans << "\n";
    }
}

void solve() {
    // Find initial pt
    for (int i = 0; i < R; i++) {
        for (int j = 0; j < C; j++) {
            if (map[i][j] == 'S') {
                visited[i][j] = true;
                q.push(pii(i, j));
            } else if (map[i][j] == '*') {
                waterQ.push(pii(i, j));
            }
        }
    }
    

    /// bfs
    for (int time = 1; ; time++) {
        // fleed
        int waterSize = waterQ.size();
        for (int i = 0; i < waterSize; i++) {
            pii curr = waterQ.front();
            waterQ.pop();

            for (int j = 0; j < 4; j++) {
                int posr = curr.first + dr[j];
                int posc = curr.second + dc[j];

                if (posr < 0 || posc < 0 || posr >= R || posc >= C) {
                    continue;
                }

                if (map[posr][posc] == '.') {
                    map[posr][posc] = '*';
                    waterQ.push(pii(posr, posc));
                }
            }
        }

        // 고슴도치
        int queueSize = q.size();
        for (int i = 0; i < queueSize; i++) {
            pii curr = q.front();
            q.pop();

            for (int j = 0; j < 4; j++) {
                int posr = curr.first + dr[j];
                int posc = curr.second + dc[j];

                if (posr < 0 || posc < 0 || posr >= R || posc >= C) {
                    continue;
                }

                if (map[posr][posc] == 'D') {
                    ans = time;
                    return;
                }

                if (map[posr][posc] == '.' && !visited[posr][posc]) {
                    q.push(pii(posr, posc));
                    visited[posr][posc] = true;
                }
            }
        }

        if (q.size() == 0) {
            return;    
        }
    }
}
```
