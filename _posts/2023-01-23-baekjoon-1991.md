#### 트리의 구현

배열을 이용해 간단하게 구현할 수 있고, 구조체를 사용할 수 있는데 이번 문제에서는 구조체를 사용함.

```cpp
struct tree {
    node* root;
    
    static tree create_tree(char name) {
        tree tree;
        tree.root = new node {name, NULL, NULL};
        return tree;
    }

    static node* find(node* node, char name) {
        if (node == NULL) {
            return NULL;
        }

        if (node->name == name) {
            return node;
        }

        auto leftFound = find(node->left, name);
        if (leftFound != NULL) {
            return leftFound;
        }

        return find(node->right, name);
    }
}
```

#### 트리 탐색 알고리즘

재귀함수를 이용했고, 재귀함수와 output vector의 삽입 순서만을 바꾸었음.

## Solution Code

```cpp
#include <iostream>
#include <string>
#include <vector>

using namespace std;

void output();

vector<char> v;

struct node {
    char name;
    node* left;
    node* right;
};

struct tree {
    node* root;
    
    static tree create_tree(char name) {
        tree tree;
        tree.root = new node {name, NULL, NULL};
        return tree;
    }

    static node* find(node* node, char name) {
        if (node == NULL) {
            return NULL;
        }

        if (node->name == name) {
            return node;
        }

        auto leftFound = find(node->left, name);
        if (leftFound != NULL) {
            return leftFound;
        }

        return find(node->right, name);
    }

    bool addLeft(char parent, char child) {
        auto parentNode = find(root, parent);
        if (parentNode == NULL) {
            return false;
        }

        if (parentNode->left) {
            return false;
        }

        parentNode->left = new node {child, NULL, NULL};
        return true;
    }

    bool addRight(char parent, char child) {
        auto parentNode = find(root, parent);
        if (parentNode == NULL) {
            return false;
        }

        if (parentNode->right) {
            return false;
        }

        parentNode->right = new node {child, NULL, NULL};
        return true;
    }

    static void preOrder(node* node) {
        if (!node) {
            return;
        }

        v.push_back(node->name);
        preOrder(node->left);
        preOrder(node->right);
    }

    static void inOrder(node* node) {
        if (!node) {
            return;
        }

        inOrder(node->left);
        v.push_back(node->name);
        inOrder(node->right);
    }

    static void postOrder(node* node) {
        if (!node) {
            return;
        }

        postOrder(node->left);
        postOrder(node->right);
        v.push_back(node->name);
    }
};

int N;

int main() {
    freopen("input.txt", "r", stdin);

    cin >> N;
    tree tree;
    for (int i = 0; i < N; i++) {
        char root, left, right;
        cin >> root >> left >> right;
        if (i == 0) {
            tree = tree::create_tree(root);
        }
        if (left != '.') {
            tree.addLeft(root, left);
        }
        if (right != '.') {
            tree.addRight(root, right);
        }
    }

    tree::preOrder(tree.root);
    output();
    tree::inOrder(tree.root);
    output();
    tree::postOrder(tree.root);
    output();
}

void output() {
    for (int i = 0; i < v.size(); i++) {
        cout << v[i];
    }
    cout << "\n";
    v.clear();
}
```
