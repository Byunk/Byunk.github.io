## Recap

### 접근법

네 배열 중 두배열씩 모든 경우의 수를 구하고, 투 포인터 알고리즘 사용.

### 시간복잡도

경우의 수 : 2 * O(n^2)

정렬 : 2 * O(nlogn)

투 포인터 : 2 * O(n^2) (배열의 길이가 n^2)

## Solution Code

```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#define MAX 4001

using namespace std;

void solve();

int N;
long long ans;
int num[4][MAX];
vector<int> sumA;
vector<int> sumB;

int main() {
    freopen("input.txt", "r", stdin);

    cin >> N;
    for (int i = 0; i < N; i++) {
        for (int j = 0; j < 4; j++) {
            cin >> num[j][i];
        }
    }

    solve();

    cout << ans << "\n";
}

void solve() {
    for (int i = 0; i < N; i++) {
        for (int j = 0; j < N; j++) {
            sumA.push_back(num[0][i] + num[1][j]);
            sumB.push_back(num[2][i] + num[3][j]);
        }
    }

    // Sort
    sort(sumA.begin(), sumA.end(), greater<int>());
    sort(sumB.begin(), sumB.end());

    // Two Pointers
    int ptA, ptB, sum;
    ptA = ptB = sum = 0;
    while (true) {
        if (ptA == sumA.size() || ptB == sumB.size()) {
            break;
        }

        sum = sumA[ptA] + sumB[ptB];
        if (sum > 0) {
            ptA++;
        } else if (sum < 0) {
            ptB++;
        } else {
            long long cntA, cntB;
            cntA = cntB = 1;
            while (ptA != sumA.size()-1 && sumA[ptA] == sumA[ptA+1]) {
                ptA++;
                cntA++;
            }
            while (ptB != sumB.size()-1 && sumB[ptB] == sumB[ptB+1]) {
                ptB++;
                cntB++;
            }
            ptA++;
            ptB++;
            ans += cntA * cntB;
        }
    }
}
```
