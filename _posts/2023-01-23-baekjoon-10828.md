#### Deque

기본적으로 스택은 deque primary STL을 이용해 구현되어 있으므로, 마찬가지로 deque을 이용함.

## Solution Code

```cpp
#include <iostream>
#include <vector>
#include <deque>
#include <string>

using namespace std;

void solve();

int N;
deque<int> stack;
vector<string> cmd;
deque<int> num;
vector<int> out;

int main() {
    freopen("input.txt", "r", stdin);

    cin >> N;
    for (int i = 0; i < N; i++) {
        string c;
        cin >> c;
        cmd.push_back(c);
        if (c == "push") {
            int n;
            cin >> n;
            num.push_back(n);
        }
    }

    solve();

    for (int i = 0; i < out.size(); i++) {
        cout << out[i] << "\n";
    }
}

void solve() {
    int npt = 0;
    for (int i = 0; i < N; i++) {
        if (cmd[i] == "push") {
            int n = num.front();
            num.pop_front();
            stack.push_back(n);
        } else if (cmd[i] == "pop") {
            if (stack.size() == 0) {
                out.push_back(-1);
                continue;
            }

            int n = stack.back();
            stack.pop_back();
            out.push_back(n);
        } else if (cmd[i] == "size") {
            out.push_back(stack.size());
        } else if (cmd[i] == "empty") {
            if (stack.size() == 0) {
                out.push_back(1);
            } else {
                out.push_back(0);
            }
        } else if (cmd[i] == "top") {
            if (stack.size() == 0) {
                out.push_back(-1);
                continue;
            }

            int n = stack.back();
            out.push_back(n);
        }
    }
}
```
