## Solution Code

```cpp
#include <iostream>
#include <vector>
#include <set>

using namespace std;

vector<pair<int, int> > edges;
set<int> nodes;

int main() {
    freopen("input.txt", "r", stdin);

    int cnt = 1;
    while(true) {
        while(true) {
            int u, v;
            cin >> u >> v;
            if (u == -1 && v == -1) {
                return 0;
            }
            
            if (u == 0 && v == 0) {
                break;
            }

            nodes.insert(u);
            nodes.insert(v);
            edges.push_back(make_pair(u, v));
        }

        set<int> s;
        for (int i = 0; i < edges.size(); i++) {
            s.insert(edges[i].second);
        }

        if ((nodes.size() - 1 == edges.size() && s.size() == nodes.size() - 1) || nodes.size() == 0) {
            cout << "Case " << cnt << " is a tree." << "\n";
        } else {
            cout << "Case " << cnt << " is not a tree." << "\n";
        }

        edges.clear();
        nodes.clear();
        cnt++;
    }
}
```
