## Solution Code

```cpp
#include <iostream>
#include <unordered_map>
#include <string>
#include <vector>
#include <algorithm>

using namespace std;

void solve();

unordered_map<string, int> dict;
vector<string> trees;
int cnt;

int main() {
    freopen("input.txt", "r", stdin);

    string tree;
    while(getline(cin, tree)) {
        if (dict.find(tree) == dict.end()) {
            dict[tree] = 1;
            trees.push_back(tree);
        } else {
            dict[tree]++;
        }
        cnt++;
    }

    sort(trees.begin(), trees.end());
    cout << fixed;
    cout.precision(4);
    for (int i = 0; i < trees.size(); i++) {
        cout << trees[i] << " " << dict[trees[i]] * 100 / (double) cnt << "\n";
    }
}
```
