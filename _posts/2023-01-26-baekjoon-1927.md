## Solution Code

```cpp
#include <iostream>
#include <queue>
#include <vector>

using namespace std;

int N;
priority_queue<int, vector<int>, greater<int> > minHeap;

int main() {
    freopen("input.txt", "r", stdin);
    ios_base::sync_with_stdio(false);
    cin.tie(NULL); cout.tie(NULL);
    cin >> N;
    for (int i = 0; i < N; i++) {
        int n;
        cin >> n;
        if (n == 0) {
            // Deletion
            if (minHeap.size() == 0) {
                cout << 0 << "\n";
                continue;
            }

            int out = minHeap.top();
            minHeap.pop();
            cout << out << "\n";
        } else {
            // Insertion
            minHeap.push(n);
        }
    }
}

```
