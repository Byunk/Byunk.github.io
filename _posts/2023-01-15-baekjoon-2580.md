## Solution Code

```cpp
#include <iostream>
#include <vector>
#define SIZE (9)

using namespace std;

void solve();
void dfs(int idx);
bool isPossible(int n, int x, int y);
int findSector(int x);

int map[9][9];
int ans[9][9];
bool isFinished;
typedef pair<int, int> pii;
vector<pii> blanks;

int main() {
    freopen("input.txt", "r", stdin);

    for (int i = 0; i < SIZE; i++) {
        for (int j= 0; j < SIZE; j++) {
            cin >> map[i][j];
        }
    }

    solve();

    for (int i = 0; i < SIZE; i++) {
        for (int j= 0; j < SIZE; j++) {
            cout << ans[i][j] << " ";
        }
        cout << "\n";
    }
}

void solve() {
    // 빈칸 저장
    for (int i = 0; i < SIZE; i++) {
        for (int j= 0; j < SIZE; j++) {
            if (map[i][j] == 0) {
                blanks.push_back(make_pair(i, j));
            }
        }
    }
    dfs(0);
}

void dfs(int idx) {
    // 탈출조건
    if (idx == blanks.size()) {
        isFinished = true;
        for (int i = 0; i < SIZE; i++) {
            for (int j = 0; j < SIZE; j++) {
                ans[i][j] = map[i][j];
            }
        }
        return;
    }

    pii curr = blanks[idx];
    for (int i = 1; i <= SIZE; i++) {
        if (isFinished) {
            return;
        }

        if (isPossible(i, curr.first, curr.second)) {
            map[curr.first][curr.second] = i;
            dfs(idx+1);
            map[curr.first][curr.second] = 0;
        }
    }
}

bool isPossible(int n, int x, int y) {
    // 각각의 가로줄과 세로줄에는 1부터 9까지의 숫자가 한 번씩만 나타나야 한다.
    for (int i = 0; i < SIZE; i++) {
        if (n == map[i][y]) return false;
        if (n == map[x][i]) return false;
    }

    // 굵은 선으로 구분되어 있는 3x3 정사각형 안에도 1부터 9까지의 숫자가 한 번씩만 나타나야 한다.
    int xstart = findSector(x);
    int ystart = findSector(y);
    for (int i = xstart; i <= xstart+2; i++) {
        for (int j = ystart; j <= ystart+2; j++) {
            if (n == map[i][j]) return false;
        }
    }

    return true;
}

int findSector(int x) {
    if (x <= 2) {
        return 0;
    } else if (x <= 5) {
        return 3;
    } else {
        return 6;
    }
}
```
