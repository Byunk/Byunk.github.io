#### Char vs String

값을 저장하는 배열을 char로 할것이냐 string으로 할것이냐의 고민을 많이 했는데, 결론적으로 string으로 하는것이 편한듯.

- Char

int -> char : `n + '0'`

char -> int : `n - '0'`

* 자릿수가 많아지면, char array로 고려해야함.

- String

int -> string : `to_string(n)`

string -> int : `stoi(n)`

## Solution Code

```cpp
#include <iostream>
#include <string>
#include <stack>

using namespace std;

void solve();

string input;
stack<string> s;
bool flag = false;
int ans;

int main() {
    freopen("input.txt", "r", stdin);

    cin >> input;

    solve();

    if (flag) {
        cout << 0 << "\n";
    } else {
        cout << ans << "\n";
    }
}

void solve() {
    for (int i = 0; i < input.size(); i++) {
        if (input[i] == ')') {
            int result = 0;
            if (s.size() == 0) {
                flag = true;
                return;
            }

            while (s.top() != "(" && s.top() != "[") {
                int n = stoi(s.top());
                s.pop();
                result += n;

                if (s.size() == 0) {
                    flag = true;
                    return;
                }
            }

            if (s.top() == "[") {
                flag = true;
                return;
            }

            s.pop();
            if (result == 0) {
                s.push("2");
            } else {
                s.push(to_string(2 * result));
            }
        } else if (input[i] == ']') {
            int result = 0;
            if (s.size() == 0) {
                flag = true;
                return;
            }

            while (s.top() != "(" && s.top() != "[") {
                int n = stoi(s.top());
                s.pop();
                result += n;

                if (s.size() == 0) {
                    flag = true;
                    return;
                }
            }

            if (s.top() == "(") {
                flag = true;
                return;
            }

            s.pop();
            if (result == 0) {
                s.push("3");
            } else {
                s.push(to_string(3 * result));
            }
        } else {
            s.push(string(1, input[i]));
        }
        // cout << s.top() ;
    }

    int result = 0;
    while (s.size() != 0) {
        if (s.top() == "(" || s.top() == "[") {
            flag = true;
            return;
        }
        result += stoi(s.top());
        s.pop();
    }
    ans = result;
}
```
