### Solution

```cpp
#include <iostream>

#define NMAX (1000000000)
#define ABS(x) ((x)>0 ? (x) : (-x))

using namespace std;

void numX(int x);
void pop();
void inv();
void dup();
void swp();
void add();
void sub();
void mul();
void div();
void mod();

void input();
void solve();

//고스택
long long stack[1000];
int pt;

//명령어 / 숫자
int cmd[100000];
int num[100000];
int nin[10000];
int cpt, npt;


int flag; // 0 or -1 (quit) or -2 (error)
int N;

int main() {
    // NUM 0 / POP 1 / INV 2 / DUP 3 / SWP 4 / ADD 5 / SUB 6 / MUL 7
    // DIV 8 / MOD 9
    
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    // freopen("input.txt", "r", stdin);

    while (1) {
        input();
        if (flag == -1) return 0;
        solve();
    }
}

void input() {
    char o[4];
    int x;
    cpt = npt = pt = 0;
    flag = 0;

    while (true) {
        cin >> o;

        if (o[0] == 'N') {
            cin >> x;
            cmd[cpt++] = 0;
            num[npt++] = x;
        } else if (o[0] == 'P') {
            cmd[cpt++] = 1;
        } else if (o[0] == 'I') {
            cmd[cpt++] = 2;
        } else if (o[0] == 'D' && o[1] == 'U') {
            cmd[cpt++] = 3;
        } else if (o[0] == 'S' && o[1] == 'W') {
            cmd[cpt++] = 4;
        } else if (o[0] == 'A') {
            cmd[cpt++] = 5;
        } else if (o[0] == 'S' && o[1] == 'U') {
            cmd[cpt++] = 6;
        } else if (o[0] == 'M' && o[1] == 'U') {
            cmd[cpt++] = 7;
        } else if (o[0] == 'D' && o[1] == 'I') {
            cmd[cpt++] = 8;
        } else if (o[0] == 'M' && o[1] == 'O') {
            cmd[cpt++] = 9;
        } else if (o[0] == 'E') {
            break;
        } else if (o[0] == 'Q') {
            flag = -1;
            break;
        }
    }

    if (flag != -1) {
        cin >> N;
        for (int i = 0; i < N; i++) {
            cin >> nin[i];
        }
    }
}

void solve() {
    for (int i = 0; i < N; i++) {
        pt = npt = 0;
        flag = 0;
        stack[pt++] = nin[i];

        for (int j = 0; j < cpt; j++) {
            if (cmd[j] == 0) {
                numX(num[npt++]);
            } else if (pt == 0) {
                flag = -2;
                break;
            } else if (cmd[j] == 1) {
                pop();
            } else if (cmd[j] == 2) {
                inv();
            } else if (cmd[j] == 3) {
                dup();
            } else if (pt == 1) {
				flag = -2;
				break;
			} else if (cmd[j] == 4) {
                swp();
            } else if (cmd[j] == 5) {
                add();
            } else if (cmd[j] == 6) {
                sub();
            } else if (cmd[j] == 7) {
                mul();
            } else if (cmd[j] == 8) {
                div();
            } else if (cmd[j] == 9) {
                mod();
            }
            if (flag == -2) break;
        }

        if (flag == -2 || pt != 1) {
            cout << "ERROR\n";
        } else {
            cout << stack[pt-1] << "\n";
        }
    }
    cout << "\n";
}

void numX(int x) {
    stack[pt++] = x;
}

void pop() {
    pt--;
}

void inv() {
    stack[pt-1] = -stack[pt-1];
}

void dup() {
    stack[pt] = stack[pt-1];
    pt++;
}

void swp() {
    long long temp = stack[pt-1];
    stack[pt-1] = stack[pt-2];
    stack[pt-2] = temp;
}

void add() {
    long long temp = stack[pt-2] + stack[pt-1];
    if (ABS(temp) > NMAX) {
        flag = -2;
        return;
    }
    stack[pt-2] = temp;
    pt--;
}

void sub() {    
    long long temp = stack[pt-2] - stack[pt-1];
    if (ABS(temp) > NMAX) {
        flag = -2;
        return;
    }
    stack[pt-2] = temp;
    pt--;
}

void mul() {
    long long temp = stack[pt-1] * stack[pt-2];
    if (ABS(temp) > NMAX) {
        flag = -2;
        return;
    }
    stack[pt-2] = temp;
    pt--;
}

void div() {
    long long n1 = stack[pt-2];
    long long n2 = stack[pt-1];

    if (n2 == 0) {
        flag = -2;
        return;
    }

    int negativeCnt = 0;
    if (n1 < 0) negativeCnt++;
    if (n2 < 0) negativeCnt++;

    if (negativeCnt % 2 == 0) {
        stack[pt-2] = ABS(n1) / ABS(n2);
    } else {
        stack[pt-2] = - ABS(n1) / ABS(n2);
    }
    pt--;
}

void mod() {
    long long n1 = stack[pt-2];
    long long n2 = stack[pt-1];

    if (n2 == 0) {
        flag = -2;
        return;
    }

    if (n1 < 0) {
        stack[pt-2] = -ABS(n1) % ABS(n2);
    } else {
        stack[pt-2] = ABS(n1) % ABS(n2);
    }
    pt--;
}
```
