### 접근법

10!의 순열문제를 문제 단순화를 통해 접근해야했음. 처음에는 완전탐색을 이용해 접근하였는데, 시간제한에 걸렸고 다른 방법을 이용해서 해결함.

각 알파벳의 자릿수의 총합을 저장하는 방법으로 innovative하게 문제를 해결할 수 있었음.

#### 문제 단순화

알파벳에 숫자를 할당하고, 알파벳을 숫자로 변환하는 코드를 완전탐색으로 작성하면 다음과 같다. 이 경우, 완전탐색을 이용해 조합을 찾는데 O(10^n), 각 조합에 대해 O(n)의 시간이 소요된다.

```cpp
void dfs() {
    // 알파벳에 숫자 할당
    ...
}

// 알파벳을 숫자로 변환해서 계산
void cal() {
    ...
}
```

그러나, 알파벳 배열과 자릿수의 총합을 저장하는 방법으로 시간을 크게 단축시킬 수 있다.

```cpp
// 자릿수 계산
int mul = 1;
for (int i = word.size() - 1; i >= 0; i--) {
    alphabet[word[i] - 'A'] += mul;
    mul *= 10;
}

// 자릿수가 큰 순서로 정렬
sort(alphabet, alphabet + 26, compare);

// 자릿수가 큰 순서로 숫자할당
for (int i = 9; i >= 0; i--) {
    ans += alphabet[9 - i] * i;
}
```

## Solution Code

```cpp
#include <iostream>
#include <vector>
#include <string>
#include <algorithm>

using namespace std;

void solve();
bool compare(int a, int b);

int N;
vector<string> words;
int alphabet[26];
int ans;

int main() {
    freopen("input.txt", "r", stdin);

    cin >> N;
    for (int i = 0; i < N; i++) {
        string word;
        cin >> word;
        words.push_back(word);

        int mul = 1;
        for (int i = word.size() - 1; i >= 0; i--) {
            alphabet[word[i] - 'A'] += mul;
            mul *= 10;
        }
    }

    solve();

    cout << ans << "\n";
}

void solve() {
    sort(alphabet, alphabet + 26, compare);
    for (int i = 9; i >= 0; i--) {
        ans += alphabet[9 - i] * i;
    }
}

bool compare(int a, int b) {
    if (a > b) return true;
    return false;
}
```
