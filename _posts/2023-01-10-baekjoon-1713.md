### 얻어갈 점

1. C++ Custom Type sort 방법

```cpp
typedef pair <pair <int, int>, int> pii;
vector<pii> v;

bool cmp(pii a, pii b) {
    if (a.first.second > b.first.second) {
        return true;
    } else if (a.first.second == b.first.second) {
        if (a.second > b.second) {
            return true;
        } else {
            return false;
        }
    } else {
        return false;
    }
}

sort(v.begin(), v.end(), cmp);
```

2. Nested Pair 선언

```cpp
typedef pair <pair <int, int>, int> pii;
pii p = make_pair(make_pair(1, 2), 3);

p.first.first; // 1

p.second; // 3
```


## Solution Code

```cpp
#include <iostream>
#include <vector>
#include <algorithm>

#define MAX (1001)

using namespace std;

// typedef pair <int, int> pi;
typedef pair <pair <int, int>, int> pii;

bool cmp(pii a, pii b);
bool cmpBySid(pii a, pii b);
void solve();

vector<pii> v;
int recommends[MAX];

int N, R;

int main() {
    freopen("input.txt", "r", stdin);

    cin >> N >> R;
    for (int i = 0; i < R; i++) {
        cin >> recommends[i];
    }

    solve();

    sort(v.begin(), v.end(), cmpBySid);
    for (int i = 0; i < v.size(); i++) {
        cout << v[i].first.first << " ";
    }
    cout << "\n";
}

bool cmp(pii a, pii b) {
    if (a.first.second > b.first.second) {
        return true;
    } else if (a.first.second == b.first.second) {
        if (a.second > b.second) {
            return true;
        } else {
            return false;
        }
    } else {
        return false;
    }
}

bool cmpBySid(pii a, pii b) {
    if (a.first.first < b.first.first) {
        return true;
    } else {
        return false;
    }
}

void solve() {
    for (int i = 0; i < R; i++) {
        // 1. 이미 존재하는지 확인
        bool flag = false;
        for (int j = 0; j < v.size(); j++) {
            if (v[j].first.first == recommends[i]) {
                flag = true;
                // 추천수 증가
                v[j] = make_pair(make_pair(v[j].first.first, v[j].first.second + 1), v[j].second);
                break;
            }
        }

        // 존재하지 않을경우 (사진 삭제)
        if (!flag) {
            // 사진틀이 꽉 찼는지 확인
            if (v.size() == N) {
                // 사진 삭제
                v[v.size() - 1] = make_pair(make_pair(recommends[i], 1), i);
            } else {
                // 사진 추가
                v.push_back(make_pair(make_pair(recommends[i], 1), i));
            }
        }
        // 정렬
        sort(v.begin(), v.end(), cmp);
    }
}
```
