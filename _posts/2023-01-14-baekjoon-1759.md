## Recap

### 접근법

DFS를 이용한 완전탐색

### 시간복잡도

O(n^2)

### 공간복잡도

### Missed Points

1. DFS 구현에서 각 단계에서의 parameter 및 상태조건 확인

### 얻어갈 점

1. 조합은 DFS or BFS 이용해서 해결 (시간복잡도 추산이후)

## Solution Code

```cpp
#include<iostream>
#include<set>
#include<vector>
#include<string>
#include<algorithm>
#define MAX (16)

using namespace std;

void solve();
void dfs(int idx, string word);
bool contain_two_jaeum(string word);
bool contain_moeum(string word);

int L, C;
char words[MAX];
string strWords;
set<char> moeum;
vector<string> ans;

int main() {
    freopen("input.txt", "r", stdin);

    cin >> L >> C;
    for (int i = 0; i < C; i++) {
        cin >> words[i];
    }

    solve();
    
    for (int i = 0 ; i < ans.size(); i++) {
        cout << ans[i] << "\n";
    }
}

void solve() {
    moeum.insert('a');
    moeum.insert('e');
    moeum.insert('i');
    moeum.insert('o');
    moeum.insert('u');

    // words를 정렬
    strWords = words;
    sort(strWords.begin(), strWords.end());

    dfs(0, "");
}

void dfs(int idx, string word) {
    if (idx == C) {
        if (word.size() == L && contain_two_jaeum(word) && contain_moeum(word)) {
            ans.push_back(word);
        }
        return;
    }
    
    dfs(idx + 1, word + strWords[idx]);
    dfs(idx + 1, word);
}

bool contain_two_jaeum(string word) {
    int flag = 0;
    for (int i = 0; i < word.size(); i++) {
        if (moeum.find(word[i]) == moeum.end()) {
            flag++;
        }
    }
    if (flag >= 2) return true;
    return false;
}

bool contain_moeum(string word) {
    for (int i = 0; i < word.size(); i++) {
        if (moeum.find(word[i]) != moeum.end()) {
            return true;
        }
    }
    return false;
}
```
